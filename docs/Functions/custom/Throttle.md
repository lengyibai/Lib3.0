<ShowCode>
<template #codes>

```js
export const $throttleDelay = (() => {
  let timer;
  return function (fn, delay) {
    if (!timer) {
      timer = setTimeout(() => {
        timer = null;
        fn();
      }, delay);
    }
  };
})();
```

</template>
</ShowCode>

```js
export const $throttleInstant = (() => {
  let last = 0;
  return (callback, wait = 800) => {
    let now = +new Date();
    if (now - last > wait) {
      callback();
      last = now;
    }
  };
})();
```

</template>
